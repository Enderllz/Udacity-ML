def performance_metric2(y_true, y_predict):
    """"""
    mean = np.mean(y_true)
    SSres=0
    SStot=0
    for i in range(len(y_true)):
        SSres +=(y_true[i] - y_predict[i])**2
        SStot +=(y_true[i] - mean)**2
    
    score = 1-SSres/SStot

    return score
